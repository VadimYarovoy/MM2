\begin{align*}

    \delta_2 &= \frac{\norm{\vec{x} - \tilde{\vec{x}}}_2}{\norm{\vec{x}}_2} &
    \delta_3 &= \frac{\norm{\vec{x} - \tilde{\vec{x}}}_{\infty}}{\norm{\vec{x}}_{\infty}} \\
\end{align*}
